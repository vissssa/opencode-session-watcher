# HA Lease 高可用 Leader 选举实现计划

> **For Claude:** REQUIRED SUB-SKILL: Use superpowers:executing-plans to implement this plan task-by-task.

**Goal:** 基于 GlusterFS 共享文件系统的文件时间戳 Lease 机制，实现多副本 Active-Standby 高可用，无额外中间件依赖。

**Architecture:** 在 GlusterFS 共享路径上维护 `leader.lease` JSON 文件，Leader 每隔 `renewInterval` 刷新文件时间戳续约；Standby 轮询检测 lease 超时后，通过 write-then-verify（覆盖写 + 随机 jitter + 读回验证）竞争晋升。GlusterFS 故障时 Leader 续约失败但**不降级**（宽松续约），Standby 读不到 lease 也**不抢占**（保守策略），保证故障期间仍有且仅有原 Leader 工作。`-lease-path` 为空时退化为单实例模式，完全向后兼容。

**Tech Stack:** Go 标准库（`os`、`encoding/json`、`math/rand`、`time`）、无第三方依赖。

## 数据结构与参数

### Lease 文件格式（JSON）

```json
{"holder": "10.0.0.1:12345", "acquired_at": 1746374400000, "renewed_at": 1746374410000}
```

- `holder`：实例唯一标识，格式 `<hostname-or-ip>:<pid>`
- `acquired_at`：首次获得 Leader 的 Unix 毫秒时间戳
- `renewed_at`：最近一次续约的 Unix 毫秒时间戳（Standby 依赖此字段判断超时）

### 默认参数

| 参数 | 默认值 | 说明 |
|------|--------|------|
| `LeaseTimeout` | 30s | `renewed_at` 超出此时长认为 Leader 已死 |
| `RenewInterval` | 10s | Leader 续约间隔，建议 ≤ LeaseTimeout/3 |
| `PollInterval` | 5s | Standby 检查 lease 文件间隔 |
| jitter | 50~200ms | 竞争写入前随机等待，拉开写入窗口 |
| verifyDelay | 200ms | 写入后等待稳定再验证 |

### 选举状态机

```
[Standby] ---(lease 超时 or 不存在)--> [竞争写入]
[竞争写入] ---(写入 + jitter + 读回验证 holder==自己)--> [Leader]
[竞争写入] ---(验证失败: 被别人抢占)--> [Standby]
[Leader]   ---(renewInterval ticker)--> [续约: 更新 renewed_at]
[Leader]   ---(续约失败: GlusterFS 故障)--> [warn log, 继续工作, 不降级]
[Leader]   ---(onLeader(ctx) 返回 or ctx 取消)--> [Standby]
```

```go
// internal/lease/lease.go
package lease

import (
    "context"
    "encoding/json"
    "fmt"
    "log/slog"
    "math/rand"
    "os"
    "time"
)

// leaseFile 是写入磁盘的 JSON 结构。
type leaseFile struct {
    Holder     string `json:"holder"`
    AcquiredAt int64  `json:"acquired_at"`
    RenewedAt  int64  `json:"renewed_at"`
}

// Config 包含 Lease 的运行时参数。
type Config struct {
    LeaseTimeout  time.Duration // renewed_at 超出此时长认为 Leader 已死，默认 30s
    RenewInterval time.Duration // Leader 续约间隔，默认 10s
    PollInterval  time.Duration // Standby 轮询间隔，默认 5s
}

func (c *Config) setDefaults() {
    if c.LeaseTimeout <= 0 {
        c.LeaseTimeout = 30 * time.Second
    }
    if c.RenewInterval <= 0 {
        c.RenewInterval = 10 * time.Second
    }
    if c.PollInterval <= 0 {
        c.PollInterval = 5 * time.Second
    }
}

// Lease 管理基于文件时间戳的 Leader 选举。
type Lease struct {
    path     string
    holderID string
    cfg      Config
    logger   *slog.Logger
}

// New 创建 Lease 管理器。
// leasePath 是 GlusterFS 上的共享 lease 文件路径；
// holderID 是本实例唯一标识（建议格式: hostname:pid）。
// logger 为 nil 时使用全局默认 logger。
func New(leasePath, holderID string, cfg Config, logger *slog.Logger) *Lease {
    cfg.setDefaults()
    if logger == nil {
        logger = slog.Default()
    }
    return &Lease{
        path:     leasePath,
        holderID: holderID,
        cfg:      cfg,
        logger:   logger,
    }
}

// Run 阻塞运行选举循环：
//   - Standby 状态：轮询检查 lease 是否超时，超时后尝试竞争晋升
//   - Leader 状态：调用 onLeader(ctx)，同时后台定期续约
//   - onLeader 返回后重新进入 Standby 等待下一轮选举
//   - ctx 取消时 Run 返回
func (l *Lease) Run(ctx context.Context, onLeader func(ctx context.Context)) {
    for {
        if ctx.Err() != nil {
            return
        }
        if l.waitUntilCandidate(ctx) {
            // 成功晋升，运行 onLeader 并同步续约
            l.runAsLeader(ctx, onLeader)
        }
    }
}

// waitUntilCandidate 在 Standby 状态等待，直到本实例可以尝试晋升。
// 返回 true 表示晋升成功（自己是新 Leader）；false 表示 ctx 已取消。
func (l *Lease) waitUntilCandidate(ctx context.Context) bool {
    for {
        if ctx.Err() != nil {
            return false
        }
        // 尝试读取 lease；读取失败（含 GlusterFS 故障）时保守等待，不抢占
        lf, err := l.readLease()
        if err != nil {
            if !os.IsNotExist(err) {
                // 读取失败：GlusterFS 可能故障，保守等待
                l.logger.Warn("lease: read failed, standby waiting", "error", err)
                if !sleepWithContext(ctx, l.cfg.PollInterval) {
                    return false
                }
                continue
            }
            // lease 文件不存在，可以尝试竞争
        } else {
            // lease 文件存在，检查是否超时
            age := time.Duration(nowMillis()-lf.RenewedAt) * time.Millisecond
            if age < l.cfg.LeaseTimeout {
                // Leader 仍在活跃，继续等待
                if !sleepWithContext(ctx, l.cfg.PollInterval) {
                    return false
                }
                continue
            }
            l.logger.Info("lease: lease expired, attempting takeover",
                "expired_holder", lf.Holder,
                "age", age.Round(time.Millisecond),
            )
        }

        // 尝试竞争写入
        if l.tryAcquire(ctx, lf) {
            return true
        }
        // 竞争失败，稍后重试
        if !sleepWithContext(ctx, l.cfg.PollInterval) {
            return false
        }
    }
}

// tryAcquire 执行 write-then-verify 竞争写入：
//  1. 覆盖写入 lease，holder 设为自己
//  2. 随机 jitter 等待（50-200ms），拉开与其他竞争者的写入窗口
//  3. 读回验证 holder == 自己
//
// 返回 true 表示本实例赢得竞争；false 表示被其他实例抢占或 ctx 取消。
func (l *Lease) tryAcquire(ctx context.Context, prev *leaseFile) bool {
    now := nowMillis()
    acquiredAt := now
    if prev != nil {
        acquiredAt = prev.AcquiredAt // 保留原始 acquired_at，便于审计
    }
    lf := leaseFile{
        Holder:     l.holderID,
        AcquiredAt: acquiredAt,
        RenewedAt:  now,
    }
    if err := l.writeLease(lf); err != nil {
        l.logger.Warn("lease: write failed during acquire", "error", err)
        return false
    }

    // 随机 jitter：50-200ms，让多个竞争者的写入时间错开
    jitter := time.Duration(50+rand.Intn(151)) * time.Millisecond
    if !sleepWithContext(ctx, jitter) {
        return false
    }
    // 再等 200ms 让文件系统同步稳定
    if !sleepWithContext(ctx, 200*time.Millisecond) {
        return false
    }

    // 读回验证
    got, err := l.readLease()
    if err != nil {
        l.logger.Warn("lease: verify read failed after acquire", "error", err)
        return false
    }
    if got.Holder != l.holderID {
        l.logger.Info("lease: lost election to another instance", "winner", got.Holder)
        return false
    }
    l.logger.Info("lease: acquired leadership", "holder", l.holderID)
    return true
}

// runAsLeader 在 Leader 状态运行 onLeader，同时后台定期续约。
// onLeader 通过 leaderCtx 感知失去 Leader 资格（目前仅 ctx 取消时触发）。
// 续约失败时记录 warn 日志但不降级，符合"GlusterFS 故障时继续工作"的要求。
func (l *Lease) runAsLeader(ctx context.Context, onLeader func(ctx context.Context)) {
    leaderCtx, cancelLeader := context.WithCancel(ctx)
    defer cancelLeader()

    // 后台续约 goroutine
    renewDone := make(chan struct{})
    go func() {
        defer close(renewDone)
        ticker := time.NewTicker(l.cfg.RenewInterval)
        defer ticker.Stop()
        for {
            select {
            case <-leaderCtx.Done():
                return
            case <-ticker.C:
                if err := l.renew(); err != nil {
                    // 宽松续约：失败时 warn，不降级
                    l.logger.Warn("lease: renew failed, continuing as leader", "error", err)
                } else {
                    l.logger.Debug("lease: renewed", "holder", l.holderID)
                }
            }
        }
    }()

    onLeader(leaderCtx)
    cancelLeader()
    <-renewDone
    l.logger.Info("lease: released leadership", "holder", l.holderID)
}

// renew 更新 lease 文件的 renewed_at 为当前时间。
func (l *Lease) renew() error {
    lf, err := l.readLease()
    if err != nil {
        // 文件消失或读取失败时重建（GlusterFS 抖动场景）
        lf = &leaseFile{
            Holder:     l.holderID,
            AcquiredAt: nowMillis(),
        }
    }
    lf.RenewedAt = nowMillis()
    lf.Holder = l.holderID // 确保 holder 是自己
    return l.writeLease(*lf)
}

// readLease 读取并解析 lease 文件。
// 文件不存在时返回 os.ErrNotExist；JSON 解析失败时返回解析错误。
func (l *Lease) readLease() (*leaseFile, error) {
    data, err := os.ReadFile(l.path)
    if err != nil {
        return nil, err
    }
    var lf leaseFile
    if err := json.Unmarshal(data, &lf); err != nil {
        return nil, fmt.Errorf("lease: parse lease file: %w", err)
    }
    return &lf, nil
}

// writeLease 将 leaseFile 原子写入 lease 文件（先写临时文件再 rename）。
func (l *Lease) writeLease(lf leaseFile) error {
    data, err := json.Marshal(lf)
    if err != nil {
        return fmt.Errorf("lease: marshal lease: %w", err)
    }
    // 先写临时文件，再 rename，保证写入原子性
    tmp := l.path + ".tmp"
    if err := os.WriteFile(tmp, data, 0o644); err != nil {
        return fmt.Errorf("lease: write tmp file: %w", err)
    }
    if err := os.Rename(tmp, l.path); err != nil {
        _ = os.Remove(tmp)
        return fmt.Errorf("lease: rename lease file: %w", err)
    }
    return nil
}

// nowMillis 返回当前 Unix 时间戳（毫秒）。
func nowMillis() int64 {
    return time.Now().UnixMilli()
}

// sleepWithContext 等待 d 时长或 ctx 取消，返回 false 表示 ctx 已取消。
func sleepWithContext(ctx context.Context, d time.Duration) bool {
    select {
    case <-ctx.Done():
        return false
    case <-time.After(d):
        return true
    }
}
```

### Step 1: 提取 `runWatcher` 函数（重构，不改行为）

将 main.go 中从 `stateStore, err := store.Open(...)` 到末尾的 watcher 运行逻辑提取为独立函数：

```go
// runWatcher 初始化 Source/Sink/Store/Watcher 并运行同步循环（-once 或持续轮询）。
// 此函数在 Leader 上下文中被调用；ctx 取消时优雅退出。
func runWatcher(ctx context.Context, cfg config.Config, reporter *status.Reporter, logger *slog.Logger) {
    stateStore, err := store.Open(ctx, cfg.DBPath)
    if err != nil {
        logger.Error("open sqlite store failed", "error", err)
        return
    }
    defer func() {
        if err := stateStore.Close(); err != nil {
            logger.Warn("close sqlite store failed", "error", err)
        }
    }()

    sink, err := jsonlsink.NewFileSink(cfg.OutputDir, logger)
    if err != nil {
        logger.Error("open jsonl sink failed", "error", err)
        return
    }
    defer func() {
        if err := sink.Close(); err != nil {
            logger.Warn("close sink failed", "error", err)
        }
    }()

    source := opencode.NewHTTPSource(cfg.BaseURL, cfg.Timeout, logger)
    w := watcher.New(source, sink, stateStore, watcher.Config{
        MessageLimit:    cfg.MessageLimit,
        MaxMessageFetch: cfg.MaxMessageFetch,
        SessionWorkers:  cfg.SessionWorkers,
    }, logger)

    if cfg.Once {
        started := time.Now()
        result, err := w.SyncOnce(ctx)
        reporter.RecordRound(status.RoundUpdate{
            SessionsTotal:   result.SessionsTotal,
            SessionsSynced:  result.SessionsSynced,
            SessionsFailed:  result.SessionsFailed,
            MessagesNew:     result.MessagesNew,
            MaxFetchReached: result.MaxFetchReached,
            Duration:        time.Since(started),
            Err:             err,
        })
        if err != nil {
            logger.Error("sync once failed", "error", err)
        }
        return
    }

    for {
        started := time.Now()
        result, err := w.SyncOnce(ctx)
        reporter.RecordRound(status.RoundUpdate{
            SessionsTotal:   result.SessionsTotal,
            SessionsSynced:  result.SessionsSynced,
            SessionsFailed:  result.SessionsFailed,
            MessagesNew:     result.MessagesNew,
            MaxFetchReached: result.MaxFetchReached,
            Duration:        time.Since(started),
            Err:             err,
        })
        if err != nil && ctx.Err() == nil {
            logger.Error("sync round failed", "error", err)
        }
        select {
        case <-ctx.Done():
            return
        case <-timeAfter(cfg.Interval):
        }
    }
}
```

```go
// 构建 leaseID：优先使用用户指定值，否则自动生成 hostname:pid
leaseID := cfg.LeaseID

    leaseID = fmt.Sprintf("%s:%d", hostname, os.Getpid())

// -lease-path 为空时直接运行（单实例模式，向后兼容）
if cfg.LeasePath == "" {

} else {

        "lease_path", cfg.LeasePath,
        "lease_id", leaseID,

    leaseManager := lease.New(cfg.LeasePath, leaseID, leaseCfg, logger)
    leaseManager.Run(ctx, func(leaderCtx context.Context) {
        runWatcher(leaderCtx, cfg, reporter, logger)
    })
}
```
